import java.util.*;
import java.lang.*;
public class Main 
{
    String maxwordlength(String str)
    {
        int wstr=0;
        String ans="";
        for(int i=0;i<str.length();i++)
        {
            if(str.charAt(i)==' ')
            {
                String temp=str.substring(wstr,i);
                if(ans.length()<temp.length())
                {
                    ans=temp;
                }
                wstr=i+1;
            }
            else if(i==str.length()-1)
            {
                String temp=str.substring(wstr,i+1);
                if(ans.length()<temp.length())
                ans=temp;
                
            }
        }
        return ans;
    }
    public static void main(String[] args)
    {
        Scanner scan=new Scanner(System.in);
        Main obj=new Main();
        String str=scan.nextLine();
        System.out.print(obj.maxwordlength(str));
}
}
